% ---------- Simple CV macros ----------
\newcommand{\cvsection}[1]{%
  \vspace{8pt}
  {\large\color{Accent}\textbf{#1}}\par
  \vspace{2pt}
  \hrule height 0.4pt
  \vspace{6pt}
}

% Entry with title/company on left, dates on right, optional subtitle line, then bullets.
% Usage:
% \cventry{Role Title}{Company}{City, Country}{YYYY--YYYY}{
%   \item ...
% }
\newcommand{\cventry}[5]{%
  \begin{tabularx}{\textwidth}{L R}
    \textbf{#1} --- #2 & \textcolor{Muted}{#4} \\
  \end{tabularx}
  \textcolor{Muted}{#3}\par
  \vspace{2pt}
  \begin{itemize}
    #5
  \end{itemize}
}

% Compact single-line entry (e.g., Certifications)
\newcommand{\cvline}[2]{%
  \begin{tabularx}{\textwidth}{L R}
    #1 & \textcolor{Muted}{#2} \\
  \end{tabularx}
}
